#include<iostream>
using namespace std;

void dot(int (*A)[3], int (*B)[3], int (*C)[3])
{
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			for (int y = 0; y < 3; y++)
			{
				C[i][j] += A[i][y] * B[y][j];
			}
		}
		
	}

}

void multiply(int(*A)[3], int(*B)[3], int(*D)[3])
{
	for (int i = 0; i < 3; i++)
	{
		for (int y = 0; y < 3; y++)
		{
			D[i][y] += A[i][y] * B[i][y];
		}
	}
}

void Cin(int(*E)[3])
{
	for (int i = 0; i < 3; i++)
	{
		for (int y = 0; y < 3; y++)
		{
			cin >> E[i][y];
		}
	}
}

void prinf(int Arr[][3],int row ,int col)
{
	for (int i = 0; i < row; i++)
	{
		for (int j = 0; j < col; j++)
		{
			cout << Arr[i][j] << "\t";
		}
		cout << endl;
	}
}

{
	int A[3][3] ={{0}};int B[3][3] ={{0}};
	int C[3][3] = { {0} }; 	int D[3][3] = { {0} };
	cout << "请输入3×3的矩阵A: " << endl;
	Cin(A);

}
